    <script>

        document.addEventListener('DOMContentLoaded', () => {

            const dfMessenger = document.querySelector('df-messenger');

            const mainContent = document.querySelector('.content-container');


            // Listen for the chat window to open/close

            dfMessenger.addEventListener('df-chat-open-changed', (event) => {

                if (event.detail.isOpen) {

                    mainContent.style.opacity = '0'; // Fade out main content

                    mainContent.style.transform = 'scale(0.95)';

                } else {

                    mainContent.style.opacity = '1'; // Fade in main content

                    mainContent.style.transform = 'scale(1)';

                }

            });


            // Listen for responses to trigger custom actions based on bot intent

            dfMessenger.addEventListener('df-response-received', (event) => {

                const intentName = event.detail.response.queryResult.intent.displayName;

                const body = document.body;


                if (intentName === 'Stress-Relief') {

                    body.style.backgroundImage = 'linear-gradient(135deg, #c7e5e3 0%, #a4d2c7 100%)';

                    console.log('Bot is providing stress relief techniques. Website background changed.');

                } else if (intentName === 'Positive-Affirmations') {

                    body.style.backgroundImage = 'linear-gradient(135deg, #fce5d4 0%, #fbd8b9 100%)';

                    console.log('Bot is providing affirmations. Website background changed.');

                } else {

                    // Reset to default

                    body.style.backgroundImage = 'linear-gradient(135deg, #f5f7fa 0%, #c3cfe2 100%)';

                }

            });

        });

    </script>
